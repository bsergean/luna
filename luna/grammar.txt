expr                        = ( ( factor ws operator ws expr ) ( operator expr )* ) / factor
factor                      = ( "(" ws expr ws ")" ) / operand

operand                     = boolean / nil / number
operator                    = "==" / "~=" / "<" / ">" / "<=" / "=>"
                                / "+" / "-" / "*" / "/"

number                      = ~"[0-9.]+"
boolean                     = "true" / "false"
nil                         = "nil"

ws                          = ~"\s*"
